const { dispatchModules } = require("./modules");

function handleRequest(req, res) {
  const url = req.url ?? "";
  if (url === "/" || url === "") {
    res.writeHead(200, { "Content-Type": "text/plain; charset=utf-8" });
    res.end("API modular · Craft Node");
    return;
  }
  dispatchModules(req, res);
}

module.exports = { handleRequest };
