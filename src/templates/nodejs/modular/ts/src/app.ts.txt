import type { IncomingMessage, ServerResponse } from "http";
import { dispatchModules } from "./modules";

export function handleRequest(
  req: IncomingMessage,
  res: ServerResponse,
): void {
  const url = req.url ?? "";
  if (url === "/" || url === "") {
    res.writeHead(200, { "Content-Type": "text/plain; charset=utf-8" });
    res.end("API modular · Craft Node");
    return;
  }
  dispatchModules(req, res);
}
